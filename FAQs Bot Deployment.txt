🔧 server/server.js
js
Copy
Edit
const express = require('express');
const cors = require('cors');
const bodyParser = require('body-parser');
require('dotenv').config();
const { Configuration, OpenAIApi } = require('openai');

const app = express();
app.use(cors());
app.use(bodyParser.json());

const openai = new OpenAIApi(new Configuration({
  apiKey: process.env.OPENAI_API_KEY
}));

app.post('/chat', async (req, res) => {
  const { question, context } = req.body;
  const chat = await openai.createChatCompletion({
    model: 'gpt-3.5-turbo',
    messages: [
      { role: 'system', content: `Answer based on this FAQ:\n${context}` },
      { role: 'user', content: question }
    ]
  });
  res.json({ answer: chat.data.choices[0].message.content.trim() });
});

app.listen(3000, () => console.log("Server running"));

✅ STEP 3: Deploy
🟣 Backend on Render
Go to https://render.com

Click “New +” > “Web Service”

Connect your GitHub repo

Root Directory: server

Start command: npm start

Add environment variable:

Key: OPENAI_API_KEY

Value: your-key-here

Wait for deployment, copy your backend URL (https://faq-bot-api.onrender.com)

🟢 Frontend on Netlify
Go to https://netlify.com

Click “Add new site” > “Import from GitHub”

Select the same repo

Set build settings:

Directory: client

No build command needed

Netlify will give you a URL like https://faq-chatbot.netlify.app
